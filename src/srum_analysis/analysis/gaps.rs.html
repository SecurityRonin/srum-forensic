3/// Detect temporal gaps in a merged SRUM timeline.
4///
5/// Returns gap objects sorted by `start`. Two gap types:
6/// - `system_off`: all tables share the gap window.
7/// - `selective_gap`: only some tables have the gap while others have records.
8pub fn detect_gaps(all: &[serde_json::Value], threshold_hours: u64) -> Vec<serde_json::Value> {
9    use std::collections::{BTreeSet, HashMap};
10
11    let threshold_secs = (threshold_hours * 3600) as i64;
12
13    // Collect timestamps per source_table
14    let mut by_table: HashMap<String, BTreeSet<String>> = HashMap::new();
15    for v in all {
16        if let (Some(table), Some(ts)) = (
17            v.get(TABLE_KEY).and_then(|x| x.as_str()),
18            v.get("timestamp").and_then(|x| x.as_str()),
19        ) {
20            by_table
21                .entry(table.to_owned())
22                .or_default()
23                .insert(ts.to_owned());
24        }
25    }
26
27    if by_table.is_empty() {
28        return vec![];
29    }
30
31    struct Gap {
32        table: String,
33        start: String,
34        end: String,
35    }
36    let mut all_gaps: Vec<Gap> = Vec::new();
37
38    for (table, timestamps) in &by_table {
39        let ts_vec: Vec<&String> = timestamps.iter().collect();
40        for w in ts_vec.windows(2) {
41            let diff = super::iso_diff_secs(w[0], w[1]);
42            if diff >= threshold_secs {
43                all_gaps.push(Gap {
44                    table: table.clone(),
45                    start: w[0].clone(),
46                    end: w[1].clone(),
47                });
48            }
49        }
50    }
51
52    let tables: Vec<String> = by_table.keys().cloned().collect();
53
54    // Group gaps by (start, end) to determine system_off vs selective
55    let mut gap_map: HashMap<(String, String), Vec<String>> = HashMap::new();
56    for g in all_gaps {
57        gap_map
58            .entry((g.start, g.end))
59            .or_default()
60            .push(g.table);
61    }
62
63    let mut result: Vec<serde_json::Value> = Vec::new();
64
65    for ((start, end), affected_tables) in &gap_map {
66        let gap_secs = super::iso_diff_secs(start, end).max(0);
67        let gap_hours = gap_secs / 3600;
68        if affected_tables.len() == tables.len() {
69            let mut at = affected_tables.clone();
70            at.sort();
71            result.push(serde_json::json!({
72                "type": "system_off",
73                "start": start,
74                "end": end,
75                "gap_hours": gap_hours,
76                "tables_affected": at,
77            }));
78        } else {
79            for t in affected_tables {
80                result.push(serde_json::json!({
81                    "type": "selective_gap",
82                    "start": start,
83                    "end": end,
84                    "gap_hours": gap_hours,
85                    "source_table": t,
86                    "other_tables_active": true,
87                }));
88            }
89        }
90    }
91
92    // Sort by start timestamp
93    result.sort_by(|a, b| {
94        let sa = a.get("start").and_then(|v| v.as_str()).unwrap_or("");
95        let sb = b.get("start").and_then(|v| v.as_str()).unwrap_or("");
96        sa.cmp(sb)
97    });
98
99    result
100}

102/// Detect auto-increment gaps in a list of record IDs for the given table.
103///
104/// A gap means rows were deleted. Returns one object per gap range with
105/// `gap_start`, `gap_end`, `deleted_count`, and `autoinc_gap: true`.
106pub fn detect_autoinc_gaps_from_ids(table: &str, ids: &[u32]) -> Vec<serde_json::Value> {
107    let mut sorted = ids.to_vec();
108    sorted.sort_unstable();
109    let mut gaps = Vec::new();
110    for w in sorted.windows(2) {
111        if w[1] > w[0] + 1 {
112            let gap_start = w[0] + 1;
113            let gap_end = w[1] - 1;
114            let deleted_count = u64::from(gap_end - gap_start + 1);
115            gaps.push(serde_json::json!({
116                "source_table": table,
117                "gap_start": gap_start,
118                "gap_end": gap_end,
119                "deleted_count": deleted_count,
120                "autoinc_gap": true,
121            }));
122        }
123    }
124    gaps
125}

127#[cfg(test)]
128mod tests {
129    use super::*;
130    use serde_json::json;
131
132    #[test]
133    fn detect_autoinc_gaps_finds_deleted_range() {
134        let ids = vec![1u32, 2, 5, 6]; // gap at 3-4
135        let gaps = detect_autoinc_gaps_from_ids("apps", &ids);
136        assert_eq!(gaps.len(), 1);
137        assert_eq!(gaps[0]["gap_start"], json!(3u32));
138        assert_eq!(gaps[0]["gap_end"], json!(4u32));
139        assert_eq!(gaps[0]["deleted_count"], json!(2u64));
140    }
141
142    #[test]
143    fn detect_autoinc_gaps_empty_on_contiguous_ids() {
144        let ids = vec![1u32, 2, 3, 4];
145        assert!(detect_autoinc_gaps_from_ids("apps", &ids).is_empty());
146    }
