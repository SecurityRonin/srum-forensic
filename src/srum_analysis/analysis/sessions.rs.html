3/// Build user sessions from a merged SRUM timeline.
4///
5/// A session is a continuous span of `user_present == true` timestamps with
6/// gaps of ≤ 2 hours between consecutive entries. Returns an empty Vec if
7/// no user-present records exist.
8pub fn build_sessions(all: &[serde_json::Value]) -> Vec<serde_json::Value> {
9    use std::collections::{BTreeMap, BTreeSet};
10
11    let mut present_ts: BTreeSet<String> = BTreeSet::new();
12    let mut input_by_ts: BTreeMap<String, u64> = BTreeMap::new();
13
14    for v in all {
15        if let Some(ts) = v.get("timestamp").and_then(|x| x.as_str()) {
16            if v.get("user_present").and_then(|x| x.as_bool()) == Some(true) {
17                present_ts.insert(ts.to_owned());
18            }
19            if v.get(TABLE_KEY).and_then(|x| x.as_str()) == Some("apps") {
20                if let Some(ms) = v.get("user_input_time_ms").and_then(|x| x.as_u64()) {
21                    *input_by_ts.entry(ts.to_owned()).or_insert(0) += ms;
22                }
23            }
24        }
25    }
26
27    if present_ts.is_empty() {
28        return vec![];
29    }
30
31    const SESSION_GAP_SECS: i64 = 7_200; // 2 hours
32
33    let mut sessions: Vec<serde_json::Value> = Vec::new();
34    let mut iter = present_ts.iter();
35    let first = iter.next().unwrap();
36    let mut session_start = first.clone();
37    let mut session_end = first.clone();
38    let mut session_input: u64 = input_by_ts.get(first).copied().unwrap_or(0);
39
40    for ts in iter {
41        let gap = super::iso_diff_secs(&session_end, ts);
42        if gap > SESSION_GAP_SECS {
43            sessions.push(make_session(&session_start, &session_end, session_input));
44            session_start = ts.clone();
45            session_input = 0;
46        }
47        session_end = ts.clone();
48        session_input += input_by_ts.get(ts).copied().unwrap_or(0);
49    }
50    sessions.push(make_session(&session_start, &session_end, session_input));
51    sessions
52}
53
54fn make_session(start: &str, end: &str, input_ms: u64) -> serde_json::Value {
55    let duration_secs = super::iso_diff_secs(start, end).max(0);
56    let duration_hours = (duration_secs as f64 / 3600.0).ceil() as u64;
57    serde_json::json!({
58        "session_start": start,
59        "session_end": end,
60        "duration_hours": duration_hours,
61        "input_ms_total": input_ms,
62    })
63}
