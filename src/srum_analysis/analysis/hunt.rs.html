4    Exfil,
5    Miner,
6    Masquerade,
7    SuspiciousPath,
8    NoFocus,
9    Phantom,
10    Automated,
11    Beaconing,
12    NotificationC2,

16/// Filter timeline records by partial app_id (integer) or app_name (case-insensitive substring).
17pub fn filter_by_app(all: Vec<serde_json::Value>, query: &str) -> Vec<serde_json::Value> {
18    let app_lower = query.to_lowercase();
19    let app_id_filter: Option<i64> = query.parse().ok();
20    all.into_iter()
21        .filter(|v| {
22            // match by integer app_id
23            if let Some(id) = app_id_filter {
24                if v.get("app_id").and_then(|x| x.as_i64()) == Some(id) {
25                    return true;
26                }
27            }
28            // match by app_name substring (case-insensitive)
29            if let Some(name) = v.get("app_name").and_then(|x| x.as_str()) {
30                if name.to_lowercase().contains(&app_lower) {
31                    return true;
32                }
33            }
34            false
35        })
36        .collect()
37}

39/// Filter timeline records matching a named forensic hunt signature.
40pub fn hunt_filter(all: Vec<serde_json::Value>, sig: &HuntSignature) -> Vec<serde_json::Value> {
41    let flag_key: Option<&str> = match sig {
42        HuntSignature::Exfil          => Some("exfil_signal"),
43        HuntSignature::Miner          => Some("background_cpu_dominant"),
44        HuntSignature::Masquerade     => Some("masquerade_candidate"),
45        HuntSignature::SuspiciousPath => Some("suspicious_path"),
46        HuntSignature::NoFocus        => Some("no_focus_with_cpu"),
47        HuntSignature::Phantom        => Some("phantom_foreground"),
48        HuntSignature::Automated      => Some("automated_execution"),
49        HuntSignature::Beaconing      => Some("beaconing"),
50        HuntSignature::NotificationC2 => Some("notification_c2"),
51        HuntSignature::All            => None,
52    };
53
54    all.into_iter()
55        .filter(|v| match flag_key {
56            Some(key) => v.get(key).and_then(|x| x.as_bool()) == Some(true),
57            None => crate::pipeline::HEURISTIC_KEYS.iter().any(|&k| {
58                v.get(k).and_then(|x| x.as_bool()) == Some(true)
59            }),
60        })
61        .collect()
62}

64#[cfg(test)]
65mod tests {
66    use super::*;
67    use serde_json::json;

69    #[test]
70    fn filter_by_app_matches_by_id() {
71        let all = vec![
72            json!({"app_id": 42_i64}),
73            json!({"app_id": 99_i64}),
74        ];
75        let filtered = filter_by_app(all, "42");
76        assert_eq!(filtered.len(), 1);
77        assert_eq!(filtered[0]["app_id"], json!(42_i64));
78    }
79
80    #[test]
81    fn hunt_filter_automated_returns_only_flagged() {
82        let all = vec![
83            json!({"automated_execution": true}),
84            json!({"suspicious_path": true}),
85        ];
86        let filtered = hunt_filter(all, &HuntSignature::Automated);
87        assert_eq!(filtered.len(), 1);
88        assert_eq!(filtered[0]["automated_execution"], json!(true));
