1//! [`AppUsageRecord`] binary decoder — real ESE raw-tag format only.
2//!
3//! **Real ESE raw-tag layout** (`cbCommonKeyPrefix | key_suffix | col_data`):
4//! - `[0..2]`:           `cbCommonKeyPrefix` (u16 LE)
5//! - `[2..2+(KEY_LEN-pfx)]`:  key suffix bytes (KEY_LEN varies: 16 or 28 observed)
6//! - column data is always `COL_DATA_LEN` bytes at the tail of the record:
7//!   - `[col_start+0..+4]`:  ESE record header
8//!   - `[col_start+4..+8]`:  `AutoIncId` (u32 LE) → `auto_inc_id`
9//!   - `[col_start+8..+16]`: `TimeStamp` as OLE Automation Date (f64 LE) → `timestamp`
10//!   - `[col_start+16..+20]`: `AppId` (i32 LE) → `app_id`
11//!   - `[col_start+20..+24]`: `UserId` (i32 LE) → `user_id`
12//!   - `foreground_cycles` and `background_cycles` returned as 0 (columns not yet located)
13
14use srum_core::{ole_date_to_datetime, AppUsageRecord};
15
16use crate::SrumError;
17
18/// Column data is always 290 bytes at the tail, regardless of KEY_LEN variant.
19/// Verified across chainsaw (1660 records), rathbunvm_win10 (163), rathbunvm_win11 (791).
20const COL_DATA_LEN: usize = 290;
21const COL_AUTO_INC_OFF: usize = 4;
22const COL_TIMESTAMP_OFF: usize = 8;
23const COL_APP_ID_OFF: usize = 16;
24const COL_USER_ID_OFF: usize = 20;
25
26pub fn decode_app_record(data: &[u8], page: u32, tag: usize) -> Result<AppUsageRecord, SrumError> {
27    decode_real_ese(data, page, tag)
28}
29
30fn decode_real_ese(data: &[u8], page: u32, tag: usize) -> Result<AppUsageRecord, SrumError> {
31    let need = COL_DATA_LEN + 2;
32    if data.len() < need {
33        return Err(SrumError::DecodeError {
34            page,
35            tag,
36            detail: format!(
37                "app_usage ESE record too short: len={}, need >= {need}",
38                data.len()
39            ),
40        });
41    }
42    let col_start = data.len() - COL_DATA_LEN;
43
44    let auto_inc_id = u32::from_le_bytes([
45        data[col_start + COL_AUTO_INC_OFF],
46        data[col_start + COL_AUTO_INC_OFF + 1],
47        data[col_start + COL_AUTO_INC_OFF + 2],
48        data[col_start + COL_AUTO_INC_OFF + 3],
49    ]);
50
51    let ts_off = col_start + COL_TIMESTAMP_OFF;
52    let timestamp_raw = f64::from_le_bytes([
53        data[ts_off], data[ts_off + 1], data[ts_off + 2], data[ts_off + 3],
54        data[ts_off + 4], data[ts_off + 5], data[ts_off + 6], data[ts_off + 7],
55    ]);
56    let timestamp = ole_date_to_datetime(timestamp_raw);
57
58    let app_off = col_start + COL_APP_ID_OFF;
59    let app_id = i32::from_le_bytes([
60        data[app_off], data[app_off + 1], data[app_off + 2], data[app_off + 3],
61    ]);
62
63    let usr_off = col_start + COL_USER_ID_OFF;
64    let user_id = i32::from_le_bytes([
65        data[usr_off], data[usr_off + 1], data[usr_off + 2], data[usr_off + 3],
66    ]);
67
68    Ok(AppUsageRecord {
69        timestamp,
70        app_id,
71        user_id,
72        foreground_cycles: 0,
73        background_cycles: 0,
74        auto_inc_id,
75    })
76}
