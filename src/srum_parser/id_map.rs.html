1//! [`IdMapEntry`] binary decoder — real ESE raw-tag format only.
2//!
3//! **Real ESE raw-tag layout** (`cbCommonKeyPrefix | key_suffix | col_data`):
4//! - `[0..2]`:          `cbCommonKeyPrefix` (u16 LE) — bytes of full key shared with page prefix
5//! - `[2..2+(7-pfx)]`:  key suffix bytes (KEY_LEN=7 for SruDbIdMapTable)
6//! - `col_start = 2 + (7 - cb_pfx)`:
7//!   - `[col_start+0..+4]`:  ESE record header (`02 7f …`)
8//!   - `[col_start+4]`:      `IdType` (u8)   — 0=string, 1=SID
9//!   - `[col_start+5..+9]`:  `IdIndex` (i32 LE) — the FK used by other SRUM tables
10//!   - `[col_start+9..+10]`: padding byte (`00`)
11//!   - tagged section at `col_start+10`: `00 01 04 40 01` descriptor + blob bytes
12//!   - IdBlob UTF-16LE starts at `col_start+15` (only when IdType==0)
13
14use srum_core::IdMapEntry;
15
16use crate::SrumError;
17
18/// KEY_LEN for SruDbIdMapTable (7 bytes: AutoIncId u32 + IdType u8 + IdIndex i16 partial).
19const ESE_KEY_LEN: usize = 7;
20
21/// Byte offset from col_start where IdIndex (i32 LE) lives.
22const COL_ID_INDEX_OFF: usize = 5;
23
24/// Byte offset from col_start where the tagged blob section begins.
25const COL_TAGGED_OFF: usize = 10;
26
27/// Bytes consumed by the 5-byte tagged column descriptor before the blob payload.
28const TAGGED_DESCRIPTOR_LEN: usize = 5;
29
30pub fn decode_id_map_entry(data: &[u8], page: u32, tag: usize) -> Result<IdMapEntry, SrumError> {
31    if data.len() < 2 {
32        return Err(SrumError::DecodeError {
33            page,
34            tag,
35            detail: format!("id-map record too short: {}", data.len()),
36        });
37    }
38
39    let cb_pfx = u16::from_le_bytes([data[0], data[1]]) as usize;
40    if cb_pfx > ESE_KEY_LEN {
41        return Err(SrumError::DecodeError {
42            page,
43            tag,
44            detail: format!("id-map cb_pfx={cb_pfx} exceeds KEY_LEN={ESE_KEY_LEN}"),
45        });
46    }
47
48    let col_start = 2 + (ESE_KEY_LEN - cb_pfx);
49    if data.len() < col_start + 2 || data[col_start] != 0x02 || data[col_start + 1] != 0x7f {
50        return Err(SrumError::DecodeError {
51            page,
52            tag,
53            detail: format!(
54                "id-map record missing ESE header marker at col_start={col_start}: len={}",
55                data.len()
56            ),
57        });
58    }
59
60    let id_off = col_start + COL_ID_INDEX_OFF;
61    if data.len() < id_off + 4 {
62        return Err(SrumError::DecodeError {
63            page,
64            tag,
65            detail: format!(
66                "real ESE id-map too short for IdIndex: len={}, need >= {}",
67                data.len(),
68                id_off + 4
69            ),
70        });
71    }
72    let id = i32::from_le_bytes([data[id_off], data[id_off + 1], data[id_off + 2], data[id_off + 3]]);
73
74    let blob_start = col_start + COL_TAGGED_OFF + TAGGED_DESCRIPTOR_LEN;
75    let name = if data.len() > blob_start {
76        let blob = &data[blob_start..];
77        let blob = if blob.len() >= 2 && blob[blob.len() - 2] == 0 && blob[blob.len() - 1] == 0 {
78            &blob[..blob.len() - 2]
79        } else {
80            blob
81        };
82        let utf16: Vec<u16> = blob
83            .chunks_exact(2)
84            .map(|b| u16::from_le_bytes([b[0], b[1]]))
85            .collect();
86        char::decode_utf16(utf16)
87            .map(|r| r.unwrap_or(char::REPLACEMENT_CHARACTER))
88            .collect()
89    } else {
90        String::new()
91    };
92
93    Ok(IdMapEntry { id, name })
94}
