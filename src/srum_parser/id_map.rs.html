1//! [`IdMapEntry`] binary decoder.
2//!
3//! Decodes the 6+N-byte wire format used in test fixtures.
4//!
5//! Binary layout:
6//! - `[0..4]`:  `id` (i32 LE)
7//! - `[4..6]`:  `name_utf16le_byte_len` (u16 LE) — byte length of the UTF-16LE name
8//! - `[6..]`:   `name` encoded as UTF-16LE bytes
9//!
10//! This matches the real SRUDB.dat encoding where app paths are stored as
11//! UTF-16LE (the native Windows string encoding).
12
13use srum_core::{IdMapEntry, ID_MAP_MIN_SIZE};
14
15use crate::SrumError;
16
17/// Decode one `IdMapEntry` record from raw bytes.
18///
19/// # Errors
20///
21/// Returns [`SrumError::DecodeError`] if the slice is too short or the
22/// UTF-16LE name bytes contain an invalid surrogate pair.
23pub fn decode_id_map_entry(data: &[u8], page: u32, tag: usize) -> Result<IdMapEntry, SrumError> {
24    if data.len() < ID_MAP_MIN_SIZE {
25        return Err(SrumError::DecodeError {
26            page,
27            tag,
28            detail: format!(
29                "id-map record too short: {} < {ID_MAP_MIN_SIZE}",
30                data.len()
31            ),
32        });
33    }
34    let id = i32::from_le_bytes([data[0], data[1], data[2], data[3]]);
35    let name_byte_len = u16::from_le_bytes([data[4], data[5]]) as usize;
36    let total = ID_MAP_MIN_SIZE + name_byte_len;
37    if data.len() < total {
38        return Err(SrumError::DecodeError {
39            page,
40            tag,
41            detail: format!(
42                "id-map record name truncated: need {total}, got {}",
43                data.len()
44            ),
45        });
46    }
47    let name_bytes = &data[6..6 + name_byte_len];
48    let utf16_units: Vec<u16> = name_bytes
49        .chunks_exact(2)
50        .map(|b| u16::from_le_bytes([b[0], b[1]]))
51        .collect();
52    let name = char::decode_utf16(utf16_units)
53        .map(|r| r.unwrap_or(char::REPLACEMENT_CHARACTER))
54        .collect::<String>();
55    Ok(IdMapEntry { id, name })
56}
