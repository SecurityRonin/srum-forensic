1//! [`NetworkConnectivityRecord`] binary decoder — real ESE raw-tag format only.
2//!
3//! **Real ESE raw-tag layout** (`cbCommonKeyPrefix | key_suffix | col_data`):
4//! - `[0..2]`:           `cbCommonKeyPrefix` (u16 LE)
5//! - `[2..2+(28-pfx)]`:  key suffix (KEY_LEN=28 for `{DD6636C4-…}`)
6//! - `col_start = 2 + (28 - cb_pfx)`:
7//!   - `[col_start+4..+8]`:  `AutoIncId` (u32 LE)
8//!   - `[col_start+8..+16]`: `TimeStamp` as OLE Automation Date (f64 LE) → `timestamp`
9//!   - `[col_start+16..+20]`: `AppId` (i32 LE) → `app_id`
10//!   - `[col_start+20..+24]`: `UserId` (i32 LE) → `user_id`
11//!   - `[col_start+24..+32]`: `InterfaceLuid` (u64 LE) — not stored
12//!   - `[col_start+32..+36]`: `L2ProfileId` (i32 LE) → `profile_id`
13//!   - `[col_start+36..+40]`: `ConnectedTime` (u32 LE) → `connected_time`
14
15use srum_core::{ole_date_to_datetime, NetworkConnectivityRecord};
16
17use crate::SrumError;
18
19const ESE_KEY_LEN: usize = 28;
20const COL_TIMESTAMP_OFF: usize = 8;
21const COL_APP_ID_OFF: usize = 16;
22const COL_USER_ID_OFF: usize = 20;
23const COL_L2_PROFILE_ID_OFF: usize = 32;
24const COL_CONNECTED_TIME_OFF: usize = 36;
25
26pub fn decode_connectivity_record(
27    data: &[u8],
28    page: u32,
29    tag: usize,
30) -> Result<NetworkConnectivityRecord, SrumError> {
31    if data.len() < 2 {
32        return Err(SrumError::DecodeError {
33            page,
34            tag,
35            detail: format!("connectivity record too short: {}", data.len()),
36        });
37    }
38    let cb_pfx = u16::from_le_bytes([data[0], data[1]]) as usize;
39    if cb_pfx > ESE_KEY_LEN {
40        return Err(SrumError::DecodeError {
41            page,
42            tag,
43            detail: format!("connectivity cb_pfx={cb_pfx} exceeds KEY_LEN={ESE_KEY_LEN}"),
44        });
45    }
46    let col_start = 2 + (ESE_KEY_LEN - cb_pfx);
47    let need = col_start + COL_CONNECTED_TIME_OFF + 4;
48    if data.len() < need {
49        return Err(SrumError::DecodeError {
50            page,
51            tag,
52            detail: format!(
53                "connectivity real ESE record truncated: len={}, need >= {need}",
54                data.len()
55            ),
56        });
57    }
58
59    let ts_off = col_start + COL_TIMESTAMP_OFF;
60    let timestamp_raw = f64::from_le_bytes([
61        data[ts_off], data[ts_off + 1], data[ts_off + 2], data[ts_off + 3],
62        data[ts_off + 4], data[ts_off + 5], data[ts_off + 6], data[ts_off + 7],
63    ]);
64    let timestamp = ole_date_to_datetime(timestamp_raw);
65
66    let app_off = col_start + COL_APP_ID_OFF;
67    let app_id = i32::from_le_bytes([
68        data[app_off], data[app_off + 1], data[app_off + 2], data[app_off + 3],
69    ]);
70
71    let usr_off = col_start + COL_USER_ID_OFF;
72    let user_id = i32::from_le_bytes([
73        data[usr_off], data[usr_off + 1], data[usr_off + 2], data[usr_off + 3],
74    ]);
75
76    let prof_off = col_start + COL_L2_PROFILE_ID_OFF;
77    let profile_id = i32::from_le_bytes([
78        data[prof_off], data[prof_off + 1], data[prof_off + 2], data[prof_off + 3],
79    ]);
80
81    let conn_off = col_start + COL_CONNECTED_TIME_OFF;
82    let connected_time = u32::from_le_bytes([
83        data[conn_off], data[conn_off + 1], data[conn_off + 2], data[conn_off + 3],
84    ]) as u64;
85
86    Ok(NetworkConnectivityRecord {
87        timestamp,
88        app_id,
89        user_id,
90        profile_id,
91        connected_time,
92    })
93}
