1//! [`AppTimelineRecord`] binary decoder.
2//!
3//! Decodes the synthetic 32-byte wire format used in test fixtures.
4//!
5//! Binary layout (all little-endian):
6//! - `[0..8]`:   `filetime` (u64) — Windows FILETIME (100ns ticks since 1601-01-01)
7//! - `[8..12]`:  `app_id` (i32)
8//! - `[12..16]`: `user_id` (i32)
9//! - `[16..24]`: `focus_time_ms` (u64) — foreground focus duration in milliseconds
10//! - `[24..32]`: `user_input_time_ms` (u64) — user active input duration in milliseconds
11
12use srum_core::{filetime_to_datetime, AppTimelineRecord, APP_TIMELINE_RECORD_SIZE};
13
14use crate::SrumError;
15
16/// Decode one raw 32-byte record into an [`AppTimelineRecord`].
17///
18/// # Errors
19///
20/// Returns [`SrumError::DecodeError`] if `data` is shorter than 32 bytes.
21pub fn decode_app_timeline_record(
22    data: &[u8],
23    page: u32,
24    tag: usize,
25) -> Result<AppTimelineRecord, SrumError> {
26    if data.len() < APP_TIMELINE_RECORD_SIZE {
27        return Err(SrumError::DecodeError {
28            page,
29            tag,
30            detail: format!(
31                "app timeline record too short: {} < {APP_TIMELINE_RECORD_SIZE}",
32                data.len()
33            ),
34        });
35    }
36    let filetime = u64::from_le_bytes([
37        data[0], data[1], data[2], data[3], data[4], data[5], data[6], data[7],
38    ]);
39    let app_id = i32::from_le_bytes([data[8], data[9], data[10], data[11]]);
40    let user_id = i32::from_le_bytes([data[12], data[13], data[14], data[15]]);
41    let focus_time_ms = u64::from_le_bytes([
42        data[16], data[17], data[18], data[19], data[20], data[21], data[22], data[23],
43    ]);
44    let user_input_time_ms = u64::from_le_bytes([
45        data[24], data[25], data[26], data[27], data[28], data[29], data[30], data[31],
46    ]);
47    Ok(AppTimelineRecord {
48        timestamp: filetime_to_datetime(filetime),
49        app_id,
50        user_id,
51        focus_time_ms,
52        user_input_time_ms,
53    })
54}
