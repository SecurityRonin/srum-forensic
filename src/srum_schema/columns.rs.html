1//! Per-table SRUM column schema definitions.
2//!
3//! Column IDs match the ESE catalog column_id values observed in real SRUDB.dat files.
4//! Columns 1-4 are shared across all SRUM extension tables (base schema).
5//! Table-specific columns start at 5.
6
7use forensicnomicon::srum::{
8    TABLE_APP_RESOURCE_USAGE, TABLE_APP_TIMELINE, TABLE_ENERGY_USAGE, TABLE_ENERGY_USAGE_LT,
9    TABLE_ID_MAP, TABLE_NETWORK_CONNECTIVITY, TABLE_NETWORK_USAGE, TABLE_PUSH_NOTIFICATIONS,
10};
11
12/// A single SRUM column definition (compile-time static data).
13pub struct SrumColumnDef {
14    /// 1-based column identifier matching the ESE catalog column_id.
15    pub column_id: u32,
16    /// Column name as it appears in MSysObjects.
17    pub name: &'static str,
18    /// JET column type code (see ese-core coltyp constants).
19    ///
20    /// Common values: 4=LONG(i32), 15=LONG_LONG(i64), 18=UNSIGNED_LONG_LONG(u64),
21    /// 8=DATE_TIME(f64 OLE), 2=UNSIGNED_BYTE, 11=LONG_BINARY, 17=UNSIGNED_SHORT.
22    pub coltyp: u8,
23}
24

152// ── dispatch ─────────────────────────────────────────────────────────────────
153
154pub fn column_defs_for(guid: &str) -> Option<&'static [SrumColumnDef]> {
155    // Suppress "unused" warnings for BASE until a future story uses it.
156    let _ = BASE;
157    match guid {
158        x if x == TABLE_NETWORK_USAGE         => Some(NETWORK_USAGE_COLS),
159        x if x == TABLE_APP_RESOURCE_USAGE    => Some(APP_USAGE_COLS),
160        x if x == TABLE_NETWORK_CONNECTIVITY  => Some(NETWORK_CONN_COLS),
161        x if x == TABLE_ENERGY_USAGE          => Some(ENERGY_USAGE_COLS),
162        x if x == TABLE_ENERGY_USAGE_LT       => Some(ENERGY_USAGE_COLS),
163        x if x == TABLE_PUSH_NOTIFICATIONS    => Some(PUSH_NOTIF_COLS),
164        x if x == TABLE_APP_TIMELINE          => Some(APP_TIMELINE_COLS),
165        x if x == TABLE_ID_MAP                => Some(ID_MAP_COLS),
166        _ => None,
167    }
