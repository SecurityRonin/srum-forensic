1//! SRUM table and column schema lookup.
2//!
3//! Thin static-data layer exposing SRUM GUID-to-name mappings and per-table
4//! column definitions. All data is baked in at compile time — zero allocations
5//! at lookup time.
6
7mod columns;
8mod tables;
9
10pub use columns::SrumColumnDef;
11pub use tables::SrumTableInfo;
12
13/// Return the human-readable table name for a SRUM GUID (or "SruDbIdMapTable").
14///
15/// Returns `None` for unrecognised GUIDs.
16pub fn srum_table_name(guid: &str) -> Option<&'static str> {
17    tables::ALL_SRUM_TABLES
18        .iter()
19        .find(|t| t.guid == guid)
20        .map(|t| t.name)
21}
22
23/// Return the column schema for a SRUM GUID.
24///
25/// Returns `None` for unrecognised GUIDs. The slice includes all columns
26/// starting from the shared header columns (AutoIncId at 1, TimeStamp at 2,
27/// AppId at 3, UserId at 4) through to table-specific columns.
28pub fn srum_column_defs(guid: &str) -> Option<&'static [SrumColumnDef]> {
29    columns::column_defs_for(guid)
30}
31
32/// Return all known SRUM tables.
33pub fn all_srum_tables() -> &'static [SrumTableInfo] {
34    tables::ALL_SRUM_TABLES
35}
