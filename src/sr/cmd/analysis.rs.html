5/// Named forensic hunt signature for `sr hunt`.
6#[derive(clap::ValueEnum, Clone, Debug)]
7pub enum HuntSignature {
8    /// Records with exfil_signal: true (cross-table exfiltration fingerprint)
9    Exfil,
10    /// Records with background_cpu_dominant: true (miner/persistent background process)
11    Miner,
12    /// Records with masquerade_candidate: true (lookalike process name)
13    Masquerade,
14    /// Records with suspicious_path: true (execution from temp/downloads/UNC)
15    #[value(name = "suspicious-path")]
16    SuspiciousPath,
17    /// Records with no_focus_with_cpu: true (CPU without keyboard focus)
18    #[value(name = "no-focus")]
19    NoFocus,
20    /// Records with phantom_foreground: true (foreground cycles but zero focus time)
21    Phantom,
22    /// Records with automated_execution: true (focus without user input)
23    Automated,
24    /// Records with beaconing: true (regular-interval network activity)
25    Beaconing,
26    /// Records with notification_c2: true (notification-as-C2 pattern)
27    #[value(name = "notification-c2")]
28    NotificationC2,
29    /// Any record with at least one heuristic flag set
30    All,
31}

33/// Map the CLI `HuntSignature` (with Clap attrs) to `srum_analysis::analysis::HuntSignature`.
34pub fn to_analysis_sig(s: &HuntSignature) -> srum_analysis::analysis::HuntSignature {
35    use HuntSignature as C;
36    use srum_analysis::analysis::HuntSignature as A;
37    match s {
38        C::Exfil          => A::Exfil,
39        C::Miner          => A::Miner,
40        C::Masquerade     => A::Masquerade,
41        C::SuspiciousPath => A::SuspiciousPath,
42        C::NoFocus        => A::NoFocus,
43        C::Phantom        => A::Phantom,
44        C::Automated      => A::Automated,
45        C::Beaconing      => A::Beaconing,
46        C::NotificationC2 => A::NotificationC2,
47        C::All            => A::All,
48    }
49}

82pub fn run_gaps(
83    path: &Path,
84    threshold_hours: u64,
85    format: &OutputFormat,
86) -> anyhow::Result<()> {
87    let all = srum_analysis::build_timeline(path, None);
88    let mut gaps = srum_analysis::analysis::detect_gaps(&all, threshold_hours);
89
90    // AutoIncId gap detection (best-effort, appended after timestamp gaps).
91    macro_rules! add_autoinc_gaps {
92        ($table:expr, $parser:expr) => {
93            if let Ok(records) = $parser(path) {
94                let ids: Vec<u32> = records.iter().map(|r| r.auto_inc_id).collect();
95                gaps.extend(srum_analysis::analysis::detect_autoinc_gaps_from_ids($table, &ids));
96            }
97        };
98    }
99    add_autoinc_gaps!("network", srum_parser::parse_network_usage);
100    add_autoinc_gaps!("apps", srum_parser::parse_app_usage);
101    add_autoinc_gaps!("energy", srum_parser::parse_energy_usage);
102
103    print_values(&gaps, format)
104}
