5/// Map a SRUM table GUID (or well-known name) to its friendly table name.
6fn guid_to_table_name(guid: &str) -> Option<&'static str> {
7    match guid {
8        "{973F5D5C-1D90-4944-BE8E-24B94231A174}" => Some("network"),
9        "{5C8CF1C7-7257-4F13-B223-970EF5939312}" => Some("apps"),
10        "{DD6636C4-8929-4683-974E-22C046A43763}" => Some("connectivity"),
11        "{FEE4E14F-02A9-4550-B5CE-5FA2DA202E37}" => Some("energy"),
12        "{FEE4E14F-02A9-4550-B5CE-5FA2DA202E37}LT" => Some("energy-lt"),
13        "{D10CA2FE-6FCF-4F6D-848E-B2E99266FA89}" => Some("notifications"),
14        "{7ACBBAA3-D029-4BE4-9A7A-0885927F1D8F}" => Some("app-timeline"),
15        "SruDbIdMapTable" => Some("idmap"),
16        "SruDbCheckpointTable" => Some("checkpoint"),
17        _ => None,
18    }
19}

21/// Derive a Windows version hint from the set of known table names present.
22fn windows_version_hint(known_tables: &[String]) -> &'static str {
23    if known_tables.iter().any(|t| t == "app-timeline") {
24        "Windows 10 1607+ (Anniversary Update) — app-timeline table present"
25    } else {
26        "Windows 8.1+ — app-timeline table absent (pre-Anniversary Update or table not created)"
27    }
28}

30/// Collect metadata about a SRUDB.dat without parsing individual records.
31fn collect_metadata(path: &Path) -> anyhow::Result<serde_json::Value> {
32    use sha2::{Digest, Sha256};
33
34    // File hash and size
35    let bytes = std::fs::read(path)?;
36    let file_size = bytes.len() as u64;
37    let hash = hex::encode(Sha256::digest(&bytes));
38    drop(bytes);
39
40    // Open ESE to enumerate tables
41    let db = ese_core::EseDatabase::open(path)?;
42    let catalog = db.catalog_entries().unwrap_or_default();
43
44    // object_type == 1 means table
45    let mut known_tables: Vec<String> = Vec::new();
46    let mut unknown_tables: Vec<String> = Vec::new();
47
48    for entry in catalog.iter().filter(|e| e.object_type == 1) {
49        match guid_to_table_name(&entry.object_name) {
50            Some(name) => known_tables.push(name.to_owned()),
51            None => unknown_tables.push(entry.object_name.clone()),
52        }
53    }
54    known_tables.sort();
55    unknown_tables.sort();
56
57    // Record counts per known table
58    let mut record_counts = serde_json::Map::new();
59    macro_rules! count_table {
60        ($name:expr, $parser:expr) => {
61            if let Ok(records) = $parser(path) {
62                record_counts.insert($name.to_owned(), records.len().into());
63            }
64        };
65    }
66    count_table!("network", srum_parser::parse_network_usage);
67    count_table!("apps", srum_parser::parse_app_usage);
68    count_table!("connectivity", srum_parser::parse_network_connectivity);
69    count_table!("energy", srum_parser::parse_energy_usage);
70    count_table!("notifications", srum_parser::parse_push_notifications);
71    count_table!("app-timeline", srum_parser::parse_app_timeline);
72    count_table!("idmap", srum_parser::parse_id_map);
73
74    // Temporal span — build timeline (no resolve), extract first/last timestamps
75    let all = srum_analysis::build_timeline(path, None);
76    let mut first_ts: Option<String> = None;
77    let mut last_ts: Option<String> = None;
78    for v in &all {
79        if let Some(ts) = v.get("timestamp").and_then(|x| x.as_str()) {
80            if first_ts.as_deref().map_or(true, |f: &str| ts < f) {
81                first_ts = Some(ts.to_owned());
82            }
83            if last_ts.as_deref().map_or(true, |l: &str| ts > l) {
84                last_ts = Some(ts.to_owned());
85            }
86        }
87    }
88
89    let temporal_span = match (first_ts, last_ts) {
90        (Some(f), Some(l)) => serde_json::json!({"first": f, "last": l}),
91        _ => serde_json::json!(null),
92    };
93
94    let hint = windows_version_hint(&known_tables);
95
96    Ok(serde_json::json!({
97        "file_path": path.to_string_lossy(),
98        "sha256": hash,
99        "file_size_bytes": file_size,
100        "known_tables": known_tables,
101        "unknown_tables": unknown_tables,
102        "record_counts": serde_json::Value::Object(record_counts),
103        "temporal_span": temporal_span,
104        "windows_version_hint": hint,
105    }))
106}

108pub fn run_compare(
109    baseline: &Path,
110    suspect: &Path,
111    resolve: bool,
112    format: &OutputFormat,
113) -> anyhow::Result<()> {
114    let id_map_baseline = resolve.then(|| srum_analysis::load_id_map(baseline));
115    let id_map_suspect = resolve.then(|| srum_analysis::load_id_map(suspect));
116    let baseline_timeline =
117        srum_analysis::build_timeline(baseline, id_map_baseline.as_ref());
118    let suspect_timeline =
119        srum_analysis::build_timeline(suspect, id_map_suspect.as_ref());
120    let baseline_stats = srum_analysis::analysis::build_stats(baseline_timeline);
121    let suspect_stats = srum_analysis::analysis::build_stats(suspect_timeline);
122    let result = srum_analysis::analysis::compare_databases(baseline_stats, suspect_stats);
123    // compare outputs a single object, not an array — handle directly.
124    match format {
125        OutputFormat::Json => println!("{}", serde_json::to_string_pretty(&result)?),
126        OutputFormat::Ndjson => println!("{}", serde_json::to_string(&result)?),
127        OutputFormat::Csv => {
128            let mut flat: Vec<serde_json::Value> = Vec::new();
129            if let Some(arr) = result.get("new_processes").and_then(|v| v.as_array()) {
130                for r in arr {
131                    let mut r = r.clone();
132                    r.as_object_mut().unwrap().insert("diff_type".into(), "new".into());
133                    flat.push(r);
134                }
135            }
136            if let Some(arr) = result.get("changed").and_then(|v| v.as_array()) {
137                for r in arr {
138                    let mut r = r.clone();
139                    r.as_object_mut()
140                        .unwrap()
141                        .insert("diff_type".into(), "changed".into());
142                    flat.push(r);
143                }
144            }
145            if let Some(arr) = result.get("departed_processes").and_then(|v| v.as_array()) {
146                for r in arr {
147                    let mut r = r.clone();
148                    r.as_object_mut()
149                        .unwrap()
150                        .insert("diff_type".into(), "departed".into());
151                    flat.push(r);
152                }
153            }
154            print!("{}", values_to_csv(&flat)?);
155        }
156    }
157    Ok(())
158}

160pub fn run_metadata(path: &Path, format: &OutputFormat) -> anyhow::Result<()> {
161    let meta = collect_metadata(path)?;
162    match format {
163        OutputFormat::Json => println!("{}", serde_json::to_string_pretty(&meta)?),
164        OutputFormat::Ndjson => println!("{}", serde_json::to_string(&meta)?),
165        OutputFormat::Csv => {
166            print!("{}", values_to_csv(&[meta])?);
167        }
168    }
169    Ok(())
170}
