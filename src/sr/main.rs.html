1//! `sr` — SRUM forensic analysis CLI.
2//!
3//! Subcommands:
4//! - `sr network <path>` — parse and print network usage records as JSON
5//! - `sr apps <path>`   — parse and print application usage records as JSON
6//! - `sr idmap <path>`  — dump the `SruDbIdMapTable` as JSON
7
8use std::path::PathBuf;
9
10use clap::{Parser, Subcommand};
11
12mod cmd;
13mod output;
14
15use cmd::analysis::HuntSignature;
16use output::OutputFormat;
17
18/// SRUM forensic analysis tool.
19///
20/// Reads SRUDB.dat (Windows System Resource Usage Monitor database) and
21/// extracts per-process network and application usage records.
22#[derive(Parser)]
23#[command(name = "sr", about = "SRUM forensic analysis tool", version)]
24struct Cli {
25    #[command(subcommand)]
26    command: Cmd,
27}
28
29#[derive(Subcommand)]
30enum Cmd {
31    /// Parse network usage records from SRUDB.dat and print as JSON.
32    ///
33    /// Records come from the {973F5D5C-1D90-4944-BE8E-24B22A728CF2} table.
34    Network {
35        /// Path to SRUDB.dat (or a forensic copy of it).
36        path: PathBuf,
37        /// Resolve `app_id` and `user_id` to names from `SruDbIdMapTable`.
38        ///
39        /// Adds `app_name` and `user_name` fields to each record.
40        #[arg(long)]
41        resolve: bool,
42        /// Output format (json or csv).
43        #[arg(long, value_enum, default_value_t)]
44        format: OutputFormat,
45    },
46    /// Parse application usage records from SRUDB.dat and print as JSON.
47    ///
48    /// Records come from the {5C8CF1C7-7257-4F13-B223-970EF5939312} table.
49    Apps {
50        /// Path to SRUDB.dat (or a forensic copy of it).
51        path: PathBuf,
52        /// Resolve `app_id` and `user_id` to names from `SruDbIdMapTable`.
53        ///
54        /// Adds `app_name` and `user_name` fields to each record.
55        #[arg(long)]
56        resolve: bool,
57        /// Output format (json or csv).
58        #[arg(long, value_enum, default_value_t)]
59        format: OutputFormat,
60    },
61    /// Dump the `SruDbIdMapTable` as JSON — resolves `app_id` / `user_id` integers
62    /// to process paths and SIDs.
63    Idmap {
64        /// Path to SRUDB.dat (or a forensic copy of it).
65        path: PathBuf,
66        /// Output format (json or csv).
67        #[arg(long, value_enum, default_value_t)]
68        format: OutputFormat,
69    },
70    /// Parse network connectivity records — L2 connection sessions per process.
71    ///
72    /// Records come from the {DD6636C4-8929-4683-974E-22C046A43763} table.
73    Connectivity {
74        /// Path to SRUDB.dat (or a forensic copy of it).
75        path: PathBuf,
76        /// Resolve `app_id`, `user_id`, and `profile_id` to names from `SruDbIdMapTable`.
77        ///
78        /// Adds `app_name`, `user_name`, and `profile_name` fields to each record.
79        #[arg(long)]
80        resolve: bool,
81        /// Output format (json or csv).
82        #[arg(long, value_enum, default_value_t)]
83        format: OutputFormat,
84    },
85    /// Parse energy usage records — battery drain and power consumption per process.
86    ///
87    /// Records come from the {FEE4E14F-02A9-4550-B5CE-5FA2DA202E37} table.
88    Energy {
89        /// Path to SRUDB.dat (or a forensic copy of it).
90        path: PathBuf,
91        /// Resolve `app_id` and `user_id` to names from `SruDbIdMapTable`.
92        ///
93        /// Adds `app_name` and `user_name` fields to each record.
94        #[arg(long)]
95        resolve: bool,
96        /// Output format (json or csv).
97        #[arg(long, value_enum, default_value_t)]
98        format: OutputFormat,
99    },
100    /// Parse energy usage long-term records — same schema, longer accumulation window.
101    ///
102    /// Records come from the {FEE4E14F-02A9-4550-B5CE-5FA2DA202E37}LT table.
103    #[command(name = "energy-lt")]
104    EnergyLt {
105        /// Path to SRUDB.dat (or a forensic copy of it).
106        path: PathBuf,
107        /// Resolve `app_id` and `user_id` to names from `SruDbIdMapTable`.
108        ///
109        /// Adds `app_name` and `user_name` fields to each record.
110        #[arg(long)]
111        resolve: bool,
112        /// Output format (json or csv).
113        #[arg(long, value_enum, default_value_t)]
114        format: OutputFormat,
115    },
116    /// Parse push notification records — app notification activity per interval.
117    ///
118    /// Records come from the {D10CA2FE-6FCF-4F6D-848E-B2E99266FA89} table.
119    Notifications {
120        /// Path to SRUDB.dat (or a forensic copy of it).
121        path: PathBuf,
122        /// Resolve `app_id` and `user_id` to names from `SruDbIdMapTable`.
123        ///
124        /// Adds `app_name` and `user_name` fields to each record.
125        #[arg(long)]
126        resolve: bool,
127        /// Output format (json or csv).
128        #[arg(long, value_enum, default_value_t)]
129        format: OutputFormat,
130    },
131    /// Parse Application Timeline records — in-focus and user-input duration per app.
132    ///
133    /// Records come from the {7ACBBAA3-D029-4BE4-9A7A-0885927F1D8F} table.
134    /// Available since Windows 10 Anniversary Update (1607).
135    #[command(name = "app-timeline")]
136    AppTimeline {
137        /// Path to SRUDB.dat (or a forensic copy of it).
138        path: PathBuf,
139        /// Resolve `app_id` and `user_id` to names from `SruDbIdMapTable`.
140        ///
141        /// Adds `app_name` and `user_name` fields to each record.
142        #[arg(long)]
143        resolve: bool,
144        /// Output format (json or csv).
145        #[arg(long, value_enum, default_value_t)]
146        format: OutputFormat,
147    },
148    /// Show all SRUM activity for a single process across all tables.
149    ///
150    /// Accepts an integer app_id or a substring of the resolved process name
151    /// (requires --resolve for name matching).
152    #[command(name = "process")]
153    Process {
154        /// App ID (integer) or name substring to filter by.
155        app: String,
156        /// Path to SRUDB.dat (or a forensic copy of it).
157        path: PathBuf,
158        /// Resolve app_id and user_id to names from SruDbIdMapTable.
159        #[arg(long)]
160        resolve: bool,
161        /// Output format (json, csv, or ndjson).
162        #[arg(long, value_enum, default_value_t)]
163        format: OutputFormat,
164    },
165    /// Aggregate per-process statistics across all SRUM tables.
166    ///
167    /// Builds a merged timeline and summarises each app's CPU cycles, bytes,
168    /// active intervals, and heuristic flags. Output sorted by flag_count desc,
169    /// then total_background_cycles desc. Best-effort: always exits 0.
170    Stats {
171        /// Path to SRUDB.dat (or a forensic copy of it).
172        path: PathBuf,
173        /// Resolve `app_id` to names from `SruDbIdMapTable`.
174        #[arg(long)]
175        resolve: bool,
176        /// Output format (json, csv, or ndjson).
177        #[arg(long, value_enum, default_value_t)]
178        format: OutputFormat,
179    },
180    /// Derive user keyboard sessions from the SRUM timeline.
181    ///
182    /// A session is a contiguous run of timestamps where `user_present: true`.
183    /// A gap > 2 hours between timestamps starts a new session. Best-effort:
184    /// always exits 0.
185    Sessions {
186        /// Path to SRUDB.dat (or a forensic copy of it).
187        path: PathBuf,
188        /// Output format (json, csv, or ndjson).
189        #[arg(long, value_enum, default_value_t)]
190        format: OutputFormat,
191    },
192    /// Detect temporal gaps in SRUM records — identifies system-off periods and
193    /// potential targeted record deletion.
194    ///
195    /// Analyses timestamps from the merged timeline to detect two kinds of
196    /// suspicious gaps:
197    ///   - `system_off`: ALL tables have a gap at the same time window.
198    ///   - `selective_gap`: Only ONE specific table has a gap while others have records.
199    ///
200    /// Best-effort: always exits 0 even for nonexistent files.
201    Gaps {
202        /// Path to SRUDB.dat (or a forensic copy of it).
203        path: PathBuf,
204        /// Minimum gap size in hours to report (default: 2).
205        #[arg(long, default_value_t = 2u64)]
206        threshold_hours: u64,
207        /// Output format (json, csv, or ndjson).
208        #[arg(long, value_enum, default_value_t)]
209        format: OutputFormat,
210    },
211    /// Hunt for specific forensic patterns across all SRUM tables.
212    ///
213    /// Filters the merged timeline to records matching a named heuristic
214    /// signature. Best-effort: always exits 0 even for missing files.
215    Hunt {
216        /// Named forensic pattern to hunt for.
217        signature: HuntSignature,
218        /// Path to SRUDB.dat.
219        path: PathBuf,
220        #[arg(long)]
221        resolve: bool,
222        #[arg(long, value_enum, default_value_t)]
223        format: OutputFormat,
224    },
225    /// Compare two SRUDB.dat files and surface what changed between them.
226    ///
227    /// Detects new processes, departed processes, and processes whose behaviour
228    /// changed (new heuristic flags, significant byte-count deltas).
229    Compare {
230        /// Baseline SRUDB.dat (before the incident).
231        baseline: PathBuf,
232        /// Suspect SRUDB.dat (after the incident).
233        suspect: PathBuf,
234        /// Resolve app_id/user_id to names for process matching.
235        #[arg(long)]
236        resolve: bool,
237        #[arg(long, value_enum, default_value_t)]
238        format: OutputFormat,
239    },
240    /// Merge all SRUM tables into a single chronological timeline.
241    ///
242    /// Reads network, apps, connectivity, energy, notifications, and focus
243    /// records, injects a `source_table` field on each entry, and sorts by
244    /// timestamp. Apps records are automatically flagged with heuristic signals.
245    /// Tables that are absent or unreadable are silently skipped.
246    Timeline {
247        /// Path to SRUDB.dat (or a forensic copy of it).
248        path: PathBuf,
249        /// Resolve `app_id` and `user_id` to names from `SruDbIdMapTable`.
250        ///
251        /// Adds `app_name` and `user_name` fields to all records that carry
252        /// those integer IDs.
253        #[arg(long)]
254        resolve: bool,
255        /// Output format (json or csv).
256        #[arg(long, value_enum, default_value_t)]
257        format: OutputFormat,
258    },
259    /// Extract metadata from a SRUDB.dat file: SHA-256 hash, table enumeration,
260    /// record counts, temporal span, and Windows version hint.
261    Metadata {
262        path: PathBuf,
263        #[arg(long, value_enum, default_value_t)]
264        format: OutputFormat,
265    },
266}
267
268fn run() -> anyhow::Result<()> {
269    let cli = Cli::parse();
270    match cli.command {
271        Cmd::Network { path, resolve, format } =>
272            cmd::tables::run_network(&path, resolve, &format),
273        Cmd::Apps { path, resolve, format } =>
274            cmd::tables::run_apps(&path, resolve, &format),
275        Cmd::Idmap { path, format } =>
276            cmd::tables::run_idmap(&path, &format),
277        Cmd::Connectivity { path, resolve, format } =>
278            cmd::tables::run_connectivity(&path, resolve, &format),
279        Cmd::Energy { path, resolve, format } =>
280            cmd::tables::run_energy(&path, resolve, &format),
281        Cmd::EnergyLt { path, resolve, format } =>
282            cmd::tables::run_energy_lt(&path, resolve, &format),
283        Cmd::Notifications { path, resolve, format } =>
284            cmd::tables::run_notifications(&path, resolve, &format),
285        Cmd::AppTimeline { path, resolve, format } =>
286            cmd::tables::run_app_timeline(&path, resolve, &format),
287        Cmd::Stats { path, resolve, format } =>
288            cmd::analysis::run_stats(&path, resolve, &format),
289        Cmd::Sessions { path, format } =>
290            cmd::analysis::run_sessions(&path, &format),
291        Cmd::Timeline { path, resolve, format } =>
292            cmd::analysis::run_timeline(&path, resolve, &format),
293        Cmd::Process { app, path, resolve, format } =>
294            cmd::analysis::run_process(&app, &path, resolve, &format),
295        Cmd::Gaps { path, threshold_hours, format } =>
296            cmd::analysis::run_gaps(&path, threshold_hours, &format),
297        Cmd::Hunt { signature, path, resolve, format } =>
298            cmd::analysis::run_hunt(&signature, &path, resolve, &format),
299        Cmd::Compare { baseline, suspect, resolve, format } =>
300            cmd::forensics::run_compare(&baseline, &suspect, resolve, &format),
301        Cmd::Metadata { path, format } =>
302            cmd::forensics::run_metadata(&path, &format),
303    }
304}
305
306fn main() {
307    if let Err(err) = run() {
308        eprintln!("error: {err:#}");
309        std::process::exit(1);
310    }
