25/// Open an ESE database file and return the parsed header.
26///
27/// # Errors
28///
29/// Returns [`EseError`] if the file cannot be read or is not a valid ESE database.
30pub fn open(path: &std::path::Path) -> Result<EseHeader, EseError> {
31    use std::io::Read as _;
32    let mut f = std::fs::File::open(path)?;
33    let mut buf = vec![0u8; EseHeader::SIZE];
34    let n = f.read(&mut buf)?;
35    if n < EseHeader::SIZE {
36        return Err(EseError::Corrupt {
37            page: 0,
38            detail: format!("file too short: need {} bytes, got {}", EseHeader::SIZE, n),
39        });
40    }
41    EseHeader::from_bytes(&buf)
42}

50    /// Build a minimal valid ESE file in a temp file.
51    ///
52    /// `extra_pages` controls how many data pages (beyond the header page)
53    /// are appended. Each data page's first byte is set to its 1-based page
54    /// number so tests can verify the correct page was returned.
55    fn make_test_ese_file(extra_pages: usize) -> NamedTempFile {
56        let page_size: usize = 4096;
57        let mut buf = vec![0u8; page_size * (1 + extra_pages)];
58        // Header page: ESE magic at offset 4
59        buf[4..8].copy_from_slice(&0x89AB_CDEF_u32.to_le_bytes());
60        // page_size field at offset 0xEC = 236
61        buf[236..240].copy_from_slice(&(u32::try_from(page_size).unwrap()).to_le_bytes());
62        // Each data page: first byte = 1-based page number for easy identification
63        for i in 0..extra_pages {
64            buf[page_size * (i + 1)] = u8::try_from(i + 1).unwrap_or(u8::MAX);
65        }
66        let mut tmp = NamedTempFile::new().expect("tempfile");
67        tmp.write_all(&buf).expect("write test ESE file");
68        tmp
69    }
