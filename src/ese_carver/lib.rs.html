1//! ESE page carving and fragmented record reconstruction.
2//!
3//! Detects records split across page boundaries ([`detect_fragments`]) and
4//! stitches them back together ([`reconstruct_fragment`]).
5//!
6//! These are raw binary operations — forensic interpretation belongs in the
7//! `RapidTriage` correlation layer.
8
9/// A pair of page indices where a record is split across a page boundary.
10///
11/// `page_a` holds the record prefix (last tag of that page) and `page_b`
12/// holds the record suffix (first data tag of that page).
13#[derive(Debug, Clone, PartialEq, Eq)]
14pub struct FragmentPair {
15    /// 1-based page number of the page containing the record prefix.
16    pub page_a: u32,
17    /// 1-based page number of the page containing the record suffix.
18    pub page_b: u32,
19    /// Byte length of the prefix (last tag of `page_a`).
20    pub prefix_len: usize,
21    /// Byte length of the suffix (first data tag of `page_b`).
22    pub suffix_len: usize,
23}
24
25/// Scan an open ESE database for records split across consecutive page
26/// boundaries.
27///
28/// Returns a [`FragmentPair`] for every adjacent page pair where the last tag
29/// of page N and the first data tag (tag 1) of page N+1 together equal
30/// `expected_size`. Pages are iterated from 1 to `page_count - 2` (page 0 is
31/// the file header and is skipped; the last page has no successor to pair with).
32pub fn detect_fragments_db(db: &ese_core::EseDatabase, expected_size: usize) -> Vec<FragmentPair> {
33    let page_count = u32::try_from(db.page_count()).unwrap_or(u32::MAX);
34    let mut pairs = Vec::new();
35
36    for page_idx in 1..page_count.saturating_sub(1) {
37        let Ok(page_a) = db.read_page(page_idx) else {
38            continue;
39        };
40        let Ok(page_b) = db.read_page(page_idx + 1) else {
41            continue;
42        };
43        let Ok(tags_a) = page_a.tags() else { continue };
44        let Ok(tags_b) = page_b.tags() else { continue };
45
46        // Last tag of page A = potential prefix fragment
47        let Some(&(_, prefix_size)) = tags_a.last() else {
48            continue;
49        };
50        // First *data* tag of page B (tag 1; tag 0 is the page header)
51        let Some(&(_, suffix_size)) = tags_b.get(1) else {
52            continue;
53        };
54
55        let prefix_len = prefix_size as usize;
56        let suffix_len = suffix_size as usize;
57
58        if prefix_len + suffix_len == expected_size {
59            pairs.push(FragmentPair {
60                page_a: page_idx,
61                page_b: page_idx + 1,
62                prefix_len,
63                suffix_len,
64            });
65        }
66    }
67    pairs
68}

70/// Scan a flat byte slice for records split across consecutive page boundaries.
71///
72/// This is the raw-bytes variant; prefer [`detect_fragments_db`] when an open
73/// [`ese_core::EseDatabase`] is available.
74///
75/// `pages` must be a multiple of `page_size` bytes. Pages are numbered from 1
76/// (page 0 is the file header and is skipped).
77pub fn detect_fragments(pages: &[u8], page_size: usize, expected_size: usize) -> Vec<FragmentPair> {
78    if page_size == 0 || pages.len() % page_size != 0 {
79        return vec![];
80    }
81    let total_pages = pages.len() / page_size;
82    let mut pairs = Vec::new();
83
84    for page_idx in 1..total_pages.saturating_sub(1) {
85        let a_start = page_idx * page_size;
86        let b_start = (page_idx + 1) * page_size;
87        let page_a = &pages[a_start..a_start + page_size];
88        let page_b = &pages[b_start..b_start + page_size];
89
90        let Some(tags_a) = parse_tags_raw(page_a, page_size) else {
91            continue;
92        };
93        let Some(tags_b) = parse_tags_raw(page_b, page_size) else {
94            continue;
95        };
96
97        let Some(&(_, prefix_len)) = tags_a.last() else {
98            continue;
99        };
100        let Some(&(_, suffix_len)) = tags_b.get(1) else {
101            continue;
102        };
103
104        if prefix_len + suffix_len == expected_size {
105            pairs.push(FragmentPair {
106                page_a: u32::try_from(page_idx).unwrap_or(u32::MAX),
107                page_b: u32::try_from(page_idx + 1).unwrap_or(u32::MAX),
108                prefix_len,
109                suffix_len,
110            });
111        }
112    }
113    pairs
114}

116fn parse_tags_raw(page: &[u8], page_size: usize) -> Option<Vec<(usize, usize)>> {
117    if page.len() < page_size || page_size < 0x28 {
118        return None;
119    }
120    // Vista+ header: available_page_tag_count at 0x22 (0x1E in pre-Vista format).
121    let tag_count = u16::from_le_bytes([page[0x22], page[0x23]]) as usize;
122    if tag_count == 0 {
123        return None;
124    }
125    let mut tags = Vec::with_capacity(tag_count);
126    for i in 0..tag_count {
127        let pos = page_size - (i + 1) * 4;
128        let raw = u32::from_le_bytes([page[pos], page[pos + 1], page[pos + 2], page[pos + 3]]);
129        // Real ESE: cb_ (size) in LOW 13 bits, ib_ (offset) in HIGH 13 bits.
130        tags.push((((raw >> 16) & 0x1FFF) as usize, (raw & 0x1FFF) as usize));
131    }
132    Some(tags)
133}

135/// Reconstruct a fragmented record from a prefix and suffix slice.
136///
137/// Returns the stitched bytes if `prefix.len() + suffix.len() == expected_size`,
138/// `None` otherwise.
139pub fn reconstruct_fragment(prefix: &[u8], suffix: &[u8], expected_size: usize) -> Option<Vec<u8>> {
140    if prefix.len() + suffix.len() != expected_size {
141        return None;
142    }
143    let mut out = Vec::with_capacity(expected_size);
144    out.extend_from_slice(prefix);
145    out.extend_from_slice(suffix);
146    Some(out)
147}
